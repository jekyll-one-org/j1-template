<script>
  const uploadArea = document.getElementById('uploadArea');
  const fileInput = document.getElementById('fileInput');
  const audioPlayer = document.getElementById('audioPlayer');
  const audio = document.getElementById('audio');
  const transcribeBtn = document.getElementById('transcribeBtn');
  const stopBtn = document.getElementById('stopBtn');
  const copyBtn = document.getElementById('copyBtn');
  const downloadBtn = document.getElementById('downloadBtn');
  const languageSelect = document.getElementById('languageSelect');
  const status = document.getElementById('status');
  const transcriptionBox = document.getElementById('transcriptionBox');
  const progress = document.getElementById('progress');
  const progressBar = document.getElementById('progressBar');
  
  // Neue Elemente für Geräteauswahl
  const audioInputSelect = document.getElementById('audioInputSelect');
  const audioOutputSelect = document.getElementById('audioOutputSelect');
  const refreshDevicesBtn = document.getElementById('refreshDevicesBtn');

  let audioBlob = null;
  let transcription = '';
  let recognition = null;
  let audioContext = null;
  let isTranscribing = false;

  // Check browser support
  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
  if (!SpeechRecognition) {
      showStatus('error', 'Speech recognition is not supported in your browser. Please use Chrome or Edge.');
  }

  // Geräte laden beim Start
  loadAudioDevices();

  // Event Listener für Geräte-Refresh
  if (refreshDevicesBtn) {
      refreshDevicesBtn.addEventListener('click', loadAudioDevices);
  }

  // Event Listener für Output-Änderung
  if (audioOutputSelect) {
      audioOutputSelect.addEventListener('change', changeAudioOutput);
  }

  // Funktion zum Laden der verfügbaren Audio-Geräte
  async function loadAudioDevices() {
      try {
          // Erst Berechtigung anfordern
          await navigator.mediaDevices.getUserMedia({ audio: true });
          
          const devices = await navigator.mediaDevices.enumerateDevices();
          
          // Audio-Eingänge (Mikrofone)
          const audioInputs = devices.filter(device => device.kind === 'audioinput');
          if (audioInputSelect) {
              audioInputSelect.innerHTML = '';
              audioInputs.forEach(device => {
                  const option = document.createElement('option');
                  option.value = device.deviceId;
                  option.textContent = device.label || `Mikrofon ${audioInputSelect.length + 1}`;
                  audioInputSelect.appendChild(option);
              });
          }
          
          // Audio-Ausgänge (Lautsprecher)
          const audioOutputs = devices.filter(device => device.kind === 'audiooutput');
          if (audioOutputSelect) {
              audioOutputSelect.innerHTML = '';
              audioOutputs.forEach(device => {
                  const option = document.createElement('option');
                  option.value = device.deviceId;
                  option.textContent = device.label || `Lautsprecher ${audioOutputSelect.length + 1}`;
                  audioOutputSelect.appendChild(option);
              });
          }
          
          showStatus('success', 'Audio-Geräte geladen');
      } catch (error) {
          console.error('Fehler beim Laden der Geräte:', error);
          showStatus('error', 'Fehler beim Laden der Audio-Geräte. Bitte Berechtigungen prüfen.');
      }
  }

  // Funktion zum Ändern des Audio-Outputs
  async function changeAudioOutput() {
      if (!audioOutputSelect || !audio.setSinkId) {
          showStatus('error', 'Audio-Output-Auswahl wird in diesem Browser nicht unterstützt.');
          return;
      }
      
      try {
          await audio.setSinkId(audioOutputSelect.value);
          showStatus('success', 'Audio-Output geändert');
      } catch (error) {
          console.error('Fehler beim Ändern des Audio-Outputs:', error);
          showStatus('error', 'Fehler beim Ändern des Audio-Outputs');
      }
  }

  // Upload area events
  uploadArea.addEventListener('click', () => fileInput.click());

  uploadArea.addEventListener('dragover', (e) => {
      e.preventDefault();
      uploadArea.classList.add('dragover');
  });

  uploadArea.addEventListener('dragleave', () => {
      uploadArea.classList.remove('dragover');
  });

  uploadArea.addEventListener('drop', (e) => {
      e.preventDefault();
      uploadArea.classList.remove('dragover');
      const files = e.dataTransfer.files;
      if (files.length > 0) {
          handleFile(files[0]);
      }
  });

  fileInput.addEventListener('change', (e) => {
      if (e.target.files.length > 0) {
          handleFile(e.target.files[0]);
      }
  });

  function handleFile(file) {
      if (!file.type.startsWith('audio/')) {
          showStatus('error', 'Please upload a valid audio file.');
          return;
      }

      audioBlob = file;
      const url = URL.createObjectURL(file);
      audio.src = url;
      audioPlayer.classList.add('active');
      transcribeBtn.disabled = false;
      transcription = '';
      transcriptionBox.textContent = '';
      transcriptionBox.classList.remove('active');
      copyBtn.disabled = true;
      downloadBtn.disabled = true;
      showStatus('success', `File loaded: ${file.name}`);
      
      // Audio-Output setzen wenn bereits ausgewählt
      if (audioOutputSelect && audioOutputSelect.value) {
          changeAudioOutput();
      }
  }

  transcribeBtn.addEventListener('click', startTranscription);
  stopBtn.addEventListener('click', stopTranscription);

  async function startTranscription() {
      if (!SpeechRecognition) return;

      transcribeBtn.disabled = true;
      stopBtn.disabled = false;
      copyBtn.disabled = true;
      downloadBtn.disabled = true;
      transcription = '';
      transcriptionBox.textContent = '';
      transcriptionBox.classList.add('active');
      isTranscribing = true;

      recognition = new SpeechRecognition();
      recognition.continuous = true;
      recognition.interimResults = true;
      recognition.lang = languageSelect.value;

      recognition.onstart = () => {
          showStatus('info', 'Transcription in progress... Play the audio file.');
          progress.classList.add('active');
      };

      recognition.onresult = (event) => {
          let interimTranscript = '';
          let finalTranscript = '';

          for (let i = event.resultIndex; i < event.results.length; i++) {
              const transcript = event.results[i][0].transcript;
              if (event.results[i].isFinal) {
                  finalTranscript += transcript + ' ';
              } else {
                  interimTranscript += transcript;
              }
          }

          if (finalTranscript) {
              transcription += finalTranscript;
          }

          transcriptionBox.textContent = transcription + interimTranscript;
          transcriptionBox.scrollTop = transcriptionBox.scrollHeight;
      };

      recognition.onerror = (event) => {
          console.error('Recognition error:', event.error);
          if (event.error !== 'aborted') {
              showStatus('error', `Error: ${event.error}`);
          }
      };

      recognition.onend = () => {
          if (isTranscribing) {
              recognition.start();
          } else {
              finishTranscription();
          }
      };

      try {
          await audio.play();
          recognition.start();
          
          audio.addEventListener('timeupdate', updateProgress);
          audio.addEventListener('ended', () => {
              stopTranscription();
          });
      } catch (error) {
          showStatus('error', 'Failed to start transcription: ' + error.message);
          transcribeBtn.disabled = false;
          stopBtn.disabled = true;
      }
  }

  function updateProgress() {
      const percent = (audio.currentTime / audio.duration) * 100;
      progressBar.style.width = percent + '%';
  }

  function stopTranscription() {
      isTranscribing = false;
      if (recognition) {
          recognition.stop();
      }
      audio.pause();
      finishTranscription();
  }

  function finishTranscription() {
      transcribeBtn.disabled = false;
      stopBtn.disabled = true;
      progress.classList.remove('active');
      
      if (transcription.trim()) {
          copyBtn.disabled = false;
          downloadBtn.disabled = false;
          showStatus('success', 'Transcription complete!');
      } else {
          showStatus('error', 'No speech detected. Please ensure the audio is playing and try again.');
      }
  }

  copyBtn.addEventListener('click', () => {
      navigator.clipboard.writeText(transcription.trim()).then(() => {
          showStatus('success', 'Text copied to clipboard!');
      }).catch(() => {
          showStatus('error', 'Failed to copy text.');
      });
  });

  downloadBtn.addEventListener('click', () => {
      const blob = new Blob([transcription.trim()], { type: 'text/plain' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = 'transcription.txt';
      a.click();
      URL.revokeObjectURL(url);
      showStatus('success', 'Transcription downloaded!');
  });

  function showStatus(type, message) {
      status.className = `status ${type}`;
      status.textContent = message;
      status.style.display = 'block';
  }
</script>
