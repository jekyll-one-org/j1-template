TTS Karaoke Highlighting refers to a feature in text-to-speech (TTS) systems
where words or phrases in displayed text are visually highlighted in real-time
as the synthesized voice reads them aloud, mimicking the style of karaoke
lyrics on screen.

This synchronization enhances reading comprehension by combining auditory
playback with visual cues, allowing users to follow along effortlessly.
Modern neural TTS engines often provide precise word-level timestamps to
enable smooth highlighting without unnatural pauses.

const logger            = log4javascript.getLogger('j1.tts.karaoke');

function loadVoices() {
    voices = speechSynthesis.getVoices();
    voiceSelect.innerHTML = '';
    
    const germanVoices  = voices.filter(v => v.lang.startsWith('de'));

    [...germanVoices, ...englishVoices, ...otherVoices].forEach((voice, index) => {

    });

    if (sourceLanguage == 'de' && germanVoices.length > 0) {
        const voiceIndex  = voices.indexOf(germanVoices[2]);
        voiceSelect.value = voiceIndex;

    }

    if (sourceLanguage == 'en' && englishVoices.length > 0) {
        const voiceIndex  = voices.indexOf(englishVoices[0]);
        voiceSelect.value = voiceIndex;
        voiceName         = englishVoices[0].name;
    }

    // if (sourceLanguage == 'en' && englishVoices.length > 0) {
    //     const voiceIndex  = voices.indexOf(germanVoices[2]);

function onBoundaryHandlerEmulatedFast() {

    // Rough estimatation: 5 chars per word
    const avgCharsPerWord               = getAverageCharsPerWord(trimedText);

    // Rough average speaking speed: ~150-200 words per minute
    const avgSpeedSpeakingWordsPerMin   = 160;

    // Adjust based on utterance.rate (default is 1)
    const avgMsPerChar      = 60 * 1000 / (avgSpeedSpeakingWordsPerMin * avgCharsPerWord); 
    const adjustedMsPerChar = avgMsPerChar / utterance.rate;

    var currentOffset       = 0;
    
    words2speak.forEach((word) => {
        const charIndex = trimedText.indexOf(word, currentOffset);
        
        if (charIndex === -1) return;
        
        // Delay is based on absolute position from start
        const wordDelay = charIndex * adjustedMsPerChar;
        
        setTimeout(() => {
            const event = {
                name: 'word',
                charIndex: charIndex,
                charLength: word.length,
                elapsedTime: performance.now(),
                utterance: utterance
            };
            
            logger.debug`utterance.onboundary: \n spoken word: ${word} at startIndex: ${charIndex}`;
            highlightWord(charIndex);
            
        }, wordDelay);
        
        currentOffset = charIndex + word.length;
    });

}

function onBoundaryHandlerEmulatedBase() {
    // Rough estimatation: 5 chars per word
    const avgCharsPerWord               = getAverageCharsPerWord(trimedText);

    // Rough average speaking speed: ~150-200 words per minute
    const avgSpeedSpeakingWordsPerMin   = 160;

    // Adjust based on utterance.rate (default is 1)
    const avgMsPerChar      = 60 * 1000 / (avgSpeedSpeakingWordsPerMin * avgCharsPerWord); 
    const adjustedMsPerChar = avgMsPerChar / utterance.rate;

    var currentOffset       = 0;
    var cumulativeDelay     = 0;

    // Emulated boundary handler
    words2speak.forEach((word) => {
        // Find the actual position of this word in the text
        const charIndex = trimedText.indexOf(word, currentOffset);

        if (charIndex === -1) {
            logger.warn`Word "${word}" not found in text after offset ${currentOffset}`;
            return;
        }

        // Calculate delay based on characters from start to this word
        const wordDelay = cumulativeDelay;

        setTimeout(() => {
            // Emulate the boundary event object
            const event = {
                name:           'word',
                charIndex:      charIndex,
                charLength:     word.length,
                elapsedTime:    performance.now(),
                utterance:      utterance
            };

            const currentWord = trimedText.substring(charIndex, charIndex + word.length);
            logger.debug(`utterance.onboundary: \n spoken word: ${currentWord} at startIndex: ${charIndex}`);

            // Trigger your highlighting logic here
            highlightWord(charIndex);

        }, wordDelay);

        // Update offset and cumulative delay for next iteration
        currentOffset   = charIndex + word.length;
        cumulativeDelay += (word.length * adjustedMsPerChar);

        // Account for spaces/whitespace between words
        // Find next non-whitespace character
        while (currentOffset < trimedText.length && /\s/.test(trimedText[currentOffset])) {
            cumulativeDelay += adjustedMsPerChar; // Add delay for whitespace
            currentOffset++;
        }
    });
}

function getAverageCharsPerWord(text) {
    // Split by whitespace and filter out empty strings
    const words = text.trim().split(/\s+/);
    
    if (words.length === 0 || words[0] === "") return 0;

    // Calculate total characters across all words
    const totalChars = words.reduce((sum, word) => sum + word.length, 0);

    // Calculate the average
    return totalChars / words.length;
}

// Pause/Resume
function togglePause() {
    if (isPaused) {
        speechSynthesis.resume();
        updateStatus('speaking', '🔊 Spreche...');
        pauseBtn.textContent = '⏸ Pause';
        isPaused = false;
    } else {
        speechSynthesis.pause();
        updateStatus('paused', '⏸ Pausiert');
        pauseBtn.textContent = '▶ Fortsetzen';
        isPaused = true;
    }
}

// Stop
function stop() {
    speechSynthesis.cancel();
    updateStatus('ready', 'Gestoppt. Bereit zum Starten');
    speakBtn.disabled = false;
    pauseBtn.disabled = true;
    stopBtn.disabled = true;
    pauseBtn.textContent = '⏸ Pause';
    isPaused = false;
    
    // Alle Highlights entfernen
    textDisplay.querySelectorAll('span').forEach(span => {
        span.classList.remove('highlight');
    });
}

// Voices beim Laden und bei Änderungen aktualisieren
loadVoices();
if (speechSynthesis.onvoiceschanged !== undefined) {
    speechSynthesis.onvoiceschanged = loadVoices;
}

// Slider Updates
rateSlider.addEventListener('input', (e) => {
    rateValue.textContent = e.target.value;
});

pitchSlider.addEventListener('input', (e) => {
    pitchValue.textContent = e.target.value;
});

// Text in Spans aufteilen
function prepareText(text) {
    // const words = text.split(/(\s+)/);
    const words = text.split(/\s+/);
    textDisplay.innerHTML = '';
    
    words.forEach((word, index) => {
        const span = document.createElement('span');
        span.textContent = word + ' '; // add single space between words
        span.dataset.index = index;
        textDisplay.appendChild(span);
    });
}

// Highlighting
function highlightWord(charIndex) {
    const spans = textDisplay.querySelectorAll('span');
    var currentPos = 0;
    
    // Alle Highlights entfernen
    spans.forEach(span => span.classList.remove('highlight'));
    
    // Das aktuelle Wort finden und highlighten
    for (var span of spans) {
        const wordLength      = span.textContent.length;
        const spanTextContent = span.textContent;

        if (charIndex >= currentPos && charIndex < currentPos + wordLength) {
            span.classList.add('highlight');
            // Sanft zum Wort scrollen
            span.scrollIntoView({ behavior: 'smooth', block: 'center' });
            break;
        }
        currentPos += wordLength; // +1  add space
    }
}

// Status Update
function updateStatus(state, text) {
    status.className = `status ${state}`;
    status.textContent = text;
}

// Speak Funktion
function speak() {

    logger.info('speak: \n  funktion is being initialized');

    if (!trimedText) return;

    // Vorherige Utterance stoppen
    speechSynthesis.cancel();
    
    prepareText(trimedText);
    
    utterance           = new SpeechSynthesisUtterance(textInputValue);
    utterance.voice     = voices[voiceSelect.value];
    utterance.rate      = parseFloat(rateSlider.value);
    utterance.pitch     = parseFloat(pitchSlider.value);

    if (onBoundaryEmulated) {
        logger.info('speak: \n use speech synthesis utterance boundary events of type: emulated');
    } else {
        logger.info('speak: \n use speech synthesis utterance boundary events of type: native');

        // speech synthesis utterance boundary events are avalable for OS voices only 
        utterance.onboundary = (event) => {
            const dummyWord = 'dummyWord';
            
            if (event.name === 'word') {
                const startIndex    = event.charIndex;
                const length        = event.charLength;
                const targetText    = event.target.text;

                // Das Wort aus dem Originaltext extrahieren
                const currentWord = targetText.substring(startIndex, startIndex + length);
                logger.debug(`utterance.onboundary: \n spoken word: ${currentWord} at startIndex: ${startIndex}`);

                // Trigger your highlighting logic here
                highlightWord(startIndex);
            }
      };
    }

    utterance.onstart = () => {
        speakBtn.disabled   = true;
        pauseBtn.disabled   = false;
        stopBtn.disabled    = false;
        isPaused            = false;
        words2speak         = trimedText.split(/\s+/);

        updateStatus('speaking', '🔊 Speaking ...');

        if (onBoundaryEmulated) {
            onBoundaryHandlerEmulatedFast();
        }
    };

    utterance.onend = () => {
        updateStatus('ready', 'Ready to start');
        speakBtn.disabled = false;
        pauseBtn.disabled = true;
        stopBtn.disabled = true;
        // Alle Highlights entfernen
        textDisplay.querySelectorAll('span').forEach(span => {
            span.classList.remove('highlight');
        });
    };

    utterance.onerror = (event) => {
        console.error('Speech error:', event);
        updateStatus('ready', 'Fehler aufgetreten');
        speakBtn.disabled = false;
        pauseBtn.disabled = true;
        stopBtn.disabled = true;
    };

    speechSynthesis.speak(utterance);
}

// Event Listeners
speakBtn.addEventListener('click', speak);
pauseBtn.addEventListener('click', togglePause);
stopBtn.addEventListener('click', stop);

</script>
