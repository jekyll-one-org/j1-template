== TTS Karaoke

Willkommen zum Text-to-Speech Karaoke Demo!

Diese Anwendung zeigt, wie man Wörter während des Sprechens hervorhebt.
Die Web Speech API verwendet Boundary Events, um die Position jedes Wortes
zu verfolgen.

<script>

const textDisplay   = document.getElementById('textDisplay');
const textInput     = document.getElementById('textInput');
const speakBtn      = document.getElementById('speakBtn');
const pauseBtn      = document.getElementById('pauseBtn');
const stopBtn       = document.getElementById('stopBtn');
const voiceSelect   = document.getElementById('voiceSelect');
const rateSlider    = document.getElementById('rateSlider');
const pitchSlider   = document.getElementById('pitchSlider');
const rateValue     = document.getElementById('rateValue');
const pitchValue    = document.getElementById('pitchValue');
const status        = document.getElementById('status');

let voices = [];
let utterance = null;
let isPaused = false;


// Voices laden
function loadVoices() {
    voices = speechSynthesis.getVoices();
    voiceSelect.innerHTML = '';
    
    // Deutsche Stimmen bevorzugen
    const germanVoices = voices.filter(v => v.lang.startsWith('de'));
    const otherVoices = voices.filter(v => !v.lang.startsWith('de'));
    
    [...germanVoices, ...otherVoices].forEach((voice, index) => {
        const option = document.createElement('option');
        option.value = index;
        option.textContent = `${voice.name} (${voice.lang})`;
        if (voice.default) option.textContent += ' - Standard';
        voiceSelect.appendChild(option);
    });

    // Erste deutsche Stimme auswählen
    if (germanVoices.length > 0) {
        const germanIndex = voices.indexOf(germanVoices[1]);
        voiceSelect.value = germanIndex;
    }
}

// Voices beim Laden und bei Änderungen aktualisieren
loadVoices();
if (speechSynthesis.onvoiceschanged !== undefined) {
    speechSynthesis.onvoiceschanged = loadVoices;
}

// Slider Updates
rateSlider.addEventListener('input', (e) => {
    rateValue.textContent = e.target.value;
});

pitchSlider.addEventListener('input', (e) => {
    pitchValue.textContent = e.target.value;
});

// Text in Spans aufteilen
function prepareText(text) {
    const words = text.split(/(\s+)/);
    textDisplay.innerHTML = '';
    
    words.forEach((word, index) => {
        const span = document.createElement('span');
        span.textContent = word;
        span.dataset.index = index;
        textDisplay.appendChild(span);
    });
}

// Highlighting
function highlightWord(charIndex) {
    const spans = textDisplay.querySelectorAll('span');
    let currentPos = 0;
    
    // Alle Highlights entfernen
    spans.forEach(span => span.classList.remove('highlight'));
    
    // Das aktuelle Wort finden und highlighten
    for (let span of spans) {
        const wordLength = span.textContent.length;
        if (charIndex >= currentPos && charIndex < currentPos + wordLength) {
            span.classList.add('highlight');
            // Sanft zum Wort scrollen
            span.scrollIntoView({ behavior: 'smooth', block: 'center' });
            break;
        }
        currentPos += wordLength;
    }
}

// Status Update
function updateStatus(state, text) {
    status.className = `status ${state}`;
    status.textContent = text;
}

// Speak Funktion
function speak() {
    const text = textInput.value.trim();
    if (!text) return;

    // Vorherige Utterance stoppen
    speechSynthesis.cancel();
    
    prepareText(text);
    
    utterance = new SpeechSynthesisUtterance(text);
    utterance.voice = voices[voiceSelect.value];
    utterance.rate = parseFloat(rateSlider.value);
    utterance.pitch = parseFloat(pitchSlider.value);

    // Boundary Event - wird bei jedem Wort ausgelöst
    utterance.onboundary = (event) => {
        if (event.name === 'word') {
            highlightWord(event.charIndex);
        }
    };

    utterance.onstart = () => {
        updateStatus('speaking', '🔊 Spreche...');
        speakBtn.disabled = true;
        pauseBtn.disabled = false;
        stopBtn.disabled = false;
        isPaused = false;
    };

    utterance.onend = () => {
        updateStatus('ready', 'Fertig! Bereit zum Starten');
        speakBtn.disabled = false;
        pauseBtn.disabled = true;
        stopBtn.disabled = true;
        // Alle Highlights entfernen
        textDisplay.querySelectorAll('span').forEach(span => {
            span.classList.remove('highlight');
        });
    };

    utterance.onerror = (event) => {
        console.error('Speech error:', event);
        updateStatus('ready', 'Fehler aufgetreten');
        speakBtn.disabled = false;
        pauseBtn.disabled = true;
        stopBtn.disabled = true;
    };

    speechSynthesis.speak(utterance);
}

// Pause/Resume
function togglePause() {
    if (isPaused) {
        speechSynthesis.resume();
        updateStatus('speaking', '🔊 Spreche...');
        pauseBtn.textContent = '⏸ Pause';
        isPaused = false;
    } else {
        speechSynthesis.pause();
        updateStatus('paused', '⏸ Pausiert');
        pauseBtn.textContent = '▶ Fortsetzen';
        isPaused = true;
    }
}

// Stop
function stop() {
    speechSynthesis.cancel();
    updateStatus('ready', 'Gestoppt. Bereit zum Starten');
    speakBtn.disabled = false;
    pauseBtn.disabled = true;
    stopBtn.disabled = true;
    pauseBtn.textContent = '⏸ Pause';
    isPaused = false;
    
    // Alle Highlights entfernen
    textDisplay.querySelectorAll('span').forEach(span => {
        span.classList.remove('highlight');
    });
}

// Event Listeners
speakBtn.addEventListener('click', speak);
pauseBtn.addEventListener('click', togglePause);
stopBtn.addEventListener('click', stop);

// Initialen Text vorbereiten
prepareText(textInput.value);

</script>
