Für Videos, die mit Video.js abgespielt werden, benötige ich Posterbilder
für die Playlist-Vorschau. Kann ich mit Hilfe von Javascript ein Bild aus dem
Video extrahieren?

Ja, das ist möglich! Du kannst mit einem versteckten `<video>`-Element und
einem `<canvas>` einen Frame aus dem Video extrahieren.

Ja, das funktioniert – mit einem versteckten `<video>`-Element und einem
`<canvas>`. Der Ablauf ist simpel: Video laden → auf den gewünschten Zeitstempel
springen → Frame auf den Canvas zeichnen → als JPEG-Data-URL exportieren.

**Der entscheidende Punkt** ist das CORS-Problem: Da dein Video von einer
externen Quelle kommt, muss der Server `Access-Control-Allow-Origin: *`
(oder deine Domain) senden, sonst wirft der Browser beim `canvas.toDataURL()`
Aufruf eine SecurityError-Exception. Das `crossOrigin = 'anonymous'`
Attribut am `<video>`-Element ist dafür zwingend nötig.

Falls der Videoserver kein CORS erlaubt, hast du drei Wege:

- **Proxy** – Video über dein eigenes Backend durchschleifen, das den Header ergänzt
- **ffmpeg serverseitig** – Poster beim Upload/Ingest einmalig vorausrechnen (empfehlenswert für große Bibliotheken)
- **Blob-URL** – Video per `fetch()` mit passenden Credentials laden und als Blob-URL übergeben, umgeht manchmal CORS-Beschränkungen


**Wie die Frame-Extraktion funktioniert**

Das Prinzip ist einfach: Ein verstecktes `<video>`-Element lädt die Datei,
wird per `currentTime` auf die gewünschte Position gesprungen, und sobald
`onseeked` feuert, zeichnet `canvas.drawImage(video, ...)` den aktuellen
Frame ins Canvas. Danach liefert `canvas.toDataURL()` das fertige Bild
als Base64-String.

```js
function captureFrame(videoEl, canvas, targetTime, outWidth) {
  return new Promise((resolve) => {
    videoEl.currentTime = targetTime;

    videoEl.onseeked = () => {
      const ctx    = canvas.getContext('2d');
      const ratio  = videoEl.videoHeight / videoEl.videoWidth;
      canvas.width  = outWidth;
      canvas.height = Math.round(outWidth * ratio);

      ctx.drawImage(videoEl, 0, 0, canvas.width, canvas.height);

      resolve(canvas.toDataURL('image/jpeg', 0.85));
    };
  });
}
```

**Wichtige Details für den produktiven Einsatz**

Das `<video>`-Element muss `crossorigin="anonymous"` tragen, sonst wirft
`toDataURL()` einen `SecurityError` bei fremden Quellen. Das Canvas bleibt
dann "tainted" und der Export schlägt still fehl.

Für den `poster`-Attribute-Workflow in deiner VideoJS-Playlist kannst du
die Data-URL direkt zuweisen:

```js
const poster = await captureFrame(vid, cnv, duration * 0.1, 640);
player.poster(poster);                       // VideoJS API
playlistItem.dataset.poster = poster;        // oder direkt am DOM-Element
```

**Grenzen**

YouTube-Videos lassen sich so nicht erfassen, weil der `<iframe>` kein
`<video>`-Element im eigenen DOM hat und der YouTube-IFrame-API kein
Canvas-Zugriff möglich ist. Für YouTube-Poster bleibt der bewährte Weg
über `//img.youtube.com/vi/{VIDEO_ID}/maxresdefault.jpg` — das zeigt auch
dein `masonry_media.yml` bereits so.

Das Tool im Widget unterstützt lokale Dateien und CORS-fähige URLs,
JPEG/PNG/WebP-Ausgabe, wählbare Breite mit automatischer Proportionserhaltung
sowie die automatische 6-Frame-Extraktion.

<script>
let vid = document.getElementById('vid');
let cnv = document.getElementById('cnv');
let ctx = cnv.getContext('2d');
let thumbCount = 0;

function switchTab(t) {
  document.getElementById('panel-file').style.display = t==='file'?'':'none';
  document.getElementById('panel-url').style.display = t==='url'?'':'none';
  document.getElementById('tab-file').className = 'vte-tab'+(t==='file'?' active':'');
  document.getElementById('tab-url').className = 'vte-tab'+(t==='url'?' active':'');
}

function onDrag(e, enter) {
  e.preventDefault();
  document.getElementById('dropzone').classList.toggle('drag-over', enter);
}

function onDrop(e) {
  e.preventDefault();
  document.getElementById('dropzone').classList.remove('drag-over');
  const f = e.dataTransfer.files[0];
  if (f && f.type.startsWith('video/')) setupVideo(URL.createObjectURL(f), f.name);
}

function loadFile(input) {
  const f = input.files[0];
  if (f) setupVideo(URL.createObjectURL(f), f.name);
}

function loadUrl() {
  const u = document.getElementById('urlInput').value.trim();
  if (u) setupVideo(u, u.split('/').pop() || 'video');
}

function setupVideo(src, name) {
  vid.src = src;
  document.getElementById('fname').textContent = name;
  vid.onloadedmetadata = () => {
    const dur = vid.duration;
    const seek = document.getElementById('seek');
    seek.max = dur;
    seek.step = dur / 1000;
    document.getElementById('vdim').textContent = vid.videoWidth + ' × ' + vid.videoHeight + ' px · ' + formatTime(dur);
    document.getElementById('controls').classList.add('show');
    seekTo(dur * 0.1);
  };
}

function onSeek(el) {
  seekTo(parseFloat(el.value));
}

function seekTo(t) {
  vid.currentTime = t;
  document.getElementById('seek').value = t;
  document.getElementById('seektime').textContent = formatTime(t);
  vid.onseeked = snapPreview;
}

function snapPreview() {}

function captureFrame() {
  const w = parseInt(document.getElementById('outW').value) || 320;
  const ratio = vid.videoHeight / vid.videoWidth;
  const h = Math.round(w * ratio);
  cnv.width = w;
  cnv.height = h;
  ctx.drawImage(vid, 0, 0, w, h);
  const fmt = document.getElementById('fmt').value;
  const qual = parseFloat(document.getElementById('qual').value);
  return { dataUrl: cnv.toDataURL(fmt, qual), time: vid.currentTime, w, h, fmt };
}

function snapCurrent() {
  const {dataUrl, time, w, h, fmt} = captureFrame();
  addThumb(dataUrl, time, w, h, fmt);
}

function snapMultiple() {
  const dur = vid.duration;
  if (!dur) return;
  const times = [0.05, 0.15, 0.30, 0.50, 0.70, 0.90].map(f => f * dur);
  let i = 0;
  function next() {
    if (i >= times.length) return;
    vid.currentTime = times[i];
    vid.onseeked = () => {
      const {dataUrl, time, w, h, fmt} = captureFrame();
      addThumb(dataUrl, time, w, h, fmt);
      i++;
      next();
    };
  }
  next();
}

function addThumb(dataUrl, time, w, h, fmt) {
  thumbCount++;
  const ext = fmt.split('/')[1];
  const grid = document.getElementById('thumbGrid');
  const div = document.createElement('div');
  div.className = 'vte-thumb';
  const img = document.createElement('img');
  img.src = dataUrl;
  img.alt = 'Frame @ ' + formatTime(time);
  const tspan = document.createElement('div');
  tspan.className = 'vte-time';
  tspan.textContent = formatTime(time) + ' · ' + w + '×' + h;
  const a = document.createElement('a');
  a.className = 'vte-dl';
  a.href = dataUrl;
  a.download = 'poster_' + formatTimeFn(time) + '.' + ext;
  a.innerHTML = '<i class="ti ti-download" aria-hidden="true"></i> Download';
  div.appendChild(img);
  div.appendChild(tspan);
  div.appendChild(a);
  grid.appendChild(div);
}

function clearThumbs() {
  document.getElementById('thumbGrid').innerHTML = '';
  thumbCount = 0;
}

function formatTime(s) {
  if (!isFinite(s)) return '0s';
  const m = Math.floor(s / 60);
  const sec = (s % 60).toFixed(1);
  return m > 0 ? m + ':' + (sec < 10 ? '0' : '') + sec : sec + 's';
}

function formatTimeFn(s) {
  return s.toFixed(1).replace('.', '_');
}
</script>
